
## 5086번 배수와 약수 

```C
#include <stdio.h>

int main()
{
    int t[10000][2], index = -1;

    do
    {
        index++;
        scanf("%d %d", &t[index][0], &t[index][1]);

    } while (t[index][0] != 0 && t[index][1] != 0);
    

    for (int i = 0; i < index; i++)
    {
        if (t[i][1] % t[i][0] == 0)
        {
            printf("factor");
        }
        else if (t[i][0] % t[i][1] == 0)
        {
            printf("multiple");
        }
        else 
        {
            printf("neither");
        }

        printf("\n");
    }

    return 0;
}
```


## 2501번 약수 구하기 

```C
#include <stdio.h>

int main()
{
    int n, k;

    scanf("%d %d", &n, &k);


    int count = 0;

    for (int i = 1; i <= n; i++)
    {
        if (n % i == 0)
        {
            count++;

            if (count == k)
            {
                printf("%d", i);
                break;
            }
        }
    }
    
    if (count < k) 
    {
        printf("0");
    }

    return 0;
}
```


## 9506번 약수들의 합 

```C
#include <stdio.h>

int main()
{
    int n, factorArr[100000] = { 0, }, sum, count;

    while (scanf("%d", &n))
    {
        if (n == -1)
        {
            break;
        }

        sum = 0;
        count = 0;

        for (int i = 1; i < n; i++)
        {
            if (n % i == 0) 
            {
                factorArr[count] = i;
                sum += i;
                count++;
            }
        }

        if (n == sum) 
        {
            printf("%d = %d", n, factorArr[0]);

            for (int i = 1; i < count; i++)
            {
                printf(" + %d", factorArr[i]);
            }
        }
        else 
        {
            printf("%d is NOT perfect.", n);
        }

        printf("\n");
    }
    
    return 0;
}
```


## 1978번 소수 찾기 

```C
#include <stdio.h>

int main()
{
    int n;

    scanf("%d", &n);

    int numArr[n];
    for (int i = 0; i < n; i++)
    {
        scanf("%d", &numArr[i]);
    }


    int count = 0, isPrime;

    for (int i = 0; i < n; i++)
    {
        if (numArr[i] > 1 && (numArr[i] % 2 != 0 || numArr[i] == 2)) 
        {
            isPrime = 1;

            for (int j = 3; j < numArr[i]; j += 2)
            {
                if (numArr[i] % j == 0)
                {
                    isPrime = 0;
                    break;
                }
            }
        } 
        else 
        {
            isPrime = 0;
        }


        if (isPrime)
        {
            count++;
        }
    }
    
    printf("%d", count);

    return 0;
}
```


## 2581번 소수 

```C
#include <stdio.h>

int main()
{
    int m, n;

    scanf("%d\n%d", &m, &n);


    int sum = 0, min = n, isPrime = 1;

    if (n > 1)
    {
        if (m < 3)
        {
            m = 3;
            sum += 2;
            min = 2;
        }
        else if (m % 2 == 0) 
        {
            m += 1;
        }

        for (int i = m; i <= n; i += 2)
        {
            for (int j = 3; j < i; j += 2)
            {
                if (i % j == 0)
                {
                    isPrime = 0;
                    break;
                }       
            }


            if (isPrime) 
            {
                sum += i;

                if (i < min)
                {
                    min = i;
                }           
            }
            else 
            {
                isPrime = 1;
            }
        }
    }


    if (sum == 0)
    {
        printf("-1");
    }
    else 
    {
        printf("%d\n%d", sum, min);
    }

    return 0;
}
```


## 11653번 소인수분해

#include <stdio.h>

{

    {
        if (n % primeNum == 0) 
        {
            n /= primeNum;
