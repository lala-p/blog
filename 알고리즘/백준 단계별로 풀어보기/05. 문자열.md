## 10809번 알파벳 찾기 

```C
#include <stdio.h>
#include <string.h>

int main()
{		
	char s[101];

	scanf("%s", s);

	int indexArr[26];
	for (int i = 0; i < 26; i++)
	{
		indexArr[i] = -1;
	}

	for (int i = 0; i < strlen(s); i++)
	{
		if (indexArr[s[i] - 'a'] == -1) {
			indexArr[s[i] - 'a'] = i;
		}
	}
	
	for (int i = 0; i < 26; i++)
	{
		printf("%d ", indexArr[i]);
	}

	return 0;
}
```


## 2675번 문자열 반복 

```C
#include <stdio.h>
#include <string.h>

int main()
{
    int t;
    
    scanf("%d", &t);
    
    int rNumArr[t];
    char sArr[t][21];
    
    for (int i = 0; i < t; i++) 
    {
        scanf("%d %s", &rNumArr[i], sArr[i]);
    }
    
    
    for (int i = 0; i < t; i++) 
    {
        for (int j = 0; j < strlen(sArr[i]); j++) 
        {
            
            for (int k = 0; k < rNumArr[i]; k++) 
            {
                printf("%c", sArr[i][j]);
            }
        }   
        
        printf("\n");
    }

    return 0;
}
```


## 1152번 단어의 개수 

```C
#include <stdio.h>
#include <string.h>

int main()
{
    char s[1000001];
    
    scanf("%[^\n]s", s);
    
    
    int wordCount = 1;
    
    for (int i = 0; i < strlen(s); i++) 
    {
        if (s[i] == ' ')
        {
            wordCount++;
        }
    }
    
    printf("%d", wordCount - (s[0] == ' ') - (s[strlen(s) - 1] == ' ') - (strlen(s) == 0));
    
    return 0;
}
```
