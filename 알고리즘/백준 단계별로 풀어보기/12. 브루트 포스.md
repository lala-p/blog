
## 2798번 블랙잭 

```C
#include <stdio.h>

int main()
{
    int n, m;

    scanf("%d %d", &n, &m);

    int numArr[n];
    for (int i = 0; i < n; i++)
    {
        scanf("%d", &numArr[i]);
    }


    int max = 0, tmp;

    for (int i = 0; i < n; i++)
    {
        if (max == m)
        {
            break;
        }

        for (int j = 0; j < n; j++)
        {
            if (j == i)
            {
                continue;
            }

            for (int k = 0; k < n; k++)
            {
                if (k == i || k == j)
                {
                    continue;
                }

                tmp = numArr[i] + numArr[j] + numArr[k];

                if (tmp <= m && tmp > max)
                {
                    max = tmp;
                }
            }
        }
    }
    
    printf("%d", max);

    return 0;
}
```


## 2231번 분해합 

```C
#include <stdio.h>

int main()
{
    int n;
    
    scanf("%d", &n);


    int sum, tmp = 0;

    for (int i = 1; i < n; i++)
    {
        sum = i;
        tmp = i;

        while (tmp > 0)
        {
            sum += tmp % 10;
            tmp /= 10;
        }
        
        if (n == sum) 
        {
            tmp = i;
            break;
        }
        else
        {
            tmp = 0;
        }
    }

    printf("%d", tmp);

    return 0;
}
```


## 19532번 수학은 비대면강의입니다 

```C
#include <stdio.h>

int main()
{
    int a, b, c, d, e, f;
    
    scanf("%d %d %d %d %d %d", &a, &b, &c, &d, &e, &f);


    int status = 0;

    for (int i = 0; i <= 999; i++)
    {
        if (status)
        {
            break;
        }
        
        for (int j = 0; j <= 999; j++)
        {
            if (a * i + b * j == c && d * i + e * j == f) 
            {
                a = i;
                b = j;
                status = 1;
                break;
            }

            if (a * -i + b * -j == c && d * -i + e * -j == f)
            {
                a = -i;
                b = -j;
                status = 1;
                break;
            }

            if (a * i + b * -j == c && d * i + e * -j == f)
            {
                a = i;
                b = -j;
                status = 1;
                break;
            }

            if (a * -i + b * j == c && d * -i + e * j == f)
            {   
                a = -i;
                b = j;
                status = 1;
                break;
            }
        }
    }

    printf("%d %d", a, b);

    return 0;
}
```
