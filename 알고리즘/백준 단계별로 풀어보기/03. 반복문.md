
## 2739번 구구단

```C

{
	int num;
```

## 25304번 영수증 

```C
#include <stdio.h>

int main()
{
    int x, n;

    scanf("%d\n%d", &x, &n);    

    int a, b, sum = 0;
    for (int i = 0; i < n; i++)
    {
        scanf("%d %d", &a, &b);
        sum += a * b;
    }
    
    if (x == sum)
    {
        printf("Yes");
    } 
    else 
    {
        printf("No");
    }

    return 0;
}
```

## 2439번 별 찍기 - 2

```C
#include <stdio.h>

int main()
{
	int n;

	scanf("%d", &n);

	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < n - (i + 1); j++)
		{
			printf(" ");	
		}

		for (int j = 0; j < i + 1; j++)
		{
			printf("*");	
		}
		
		printf("\n");
	}

	return 0;
}
```
