## 24267번 알고리즘 수업 - 알고리즘 수행 시간 6 

```C
#include <stdio.h>

int main()
{
    long n;
    
    scanf("%ld", &n);


    long count = 0, tmp = 0;

    // i -> j for문 도는 횟수
    for (int i = 1; i <= n - 2; i++)
    {   
        tmp += i; 
    }

    // j -> k for문 도는 횟수
    for (int i = n - 2; i > 0; i--)
    {
        count += tmp;
        tmp -= i;
    }

    printf("%ld\n3", count);

    return 0;
}
```
